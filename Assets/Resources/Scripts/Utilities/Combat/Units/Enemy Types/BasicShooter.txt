tree("Root")
	fallback
		tree "Attack"
		tree "MoveToPlayerRow"

tree("MoveToPlayerRow")
	while not IsInPlayerRow 
		sequence
			CheckIfAbleToAttack
			MoveTowardsTargetRow
			Wait 1.5
			CheckIfAbleToAttack


tree("Attack")
	while IsInPlayerRow
		sequence
			CheckIfAbleToAttack
			DoAttack
			Wait 1.3
			CheckIfAbleToAttack
